import random,sys

# 异或运算
def do_xor(str_a,str_b):
    return (f'(\"{str_a}\"^\"{str_b}\")',chr(ord(str_a) ^ ord(str_b)))

# 获取异或结果，保存到列表
def get_xorRes(myStr):
    elems = ""
    for i in range(32, 127):
        elems += chr(i)
    xorRes_list = []
    for i in elems:
        for j in elems:
            xorRes = do_xor(i,j)
            if xorRes[1] in myStr:
                xorRes_list.append(xorRes[0] + "==>>" + xorRes[1])
    return xorRes_list

# 将列表中的多个异或结果保存到字典
def getRes(myStr):
    xorRes_list = get_xorRes(myStr)
    res = {}
    for i in range(0,len(myStr)):
        res[i] = []
        for j in xorRes_list:
            if j[-1] == myStr[i]:
                res[i].append(j)
    return res

# 从字典的元素列表中随机选取一个
def choiceRes(myStr):
    res = getRes(myStr)
    xorspel = []
    for k in res:
        choice_v = random.choice(res[k])
        xorspel.append(choice_v)
    return xorspel

if __name__ == "__main__":
    if len(sys.argv) > 1:
        myStr = sys.argv[1]
        xorspel = choiceRes(myStr)
        for i in xorspel:print(i)
    else:
        print("Usage: python3 xorForStr.py assert")

from concurrent.futures import ThreadPoolExecutor, wait

def run(loginid,password):
    print("use function")
    print(str(loginid) + " | " + str(password))

if __name__ == "__main__":
    # thread
    thread_list= []
    # max thread = 20
    with ThreadPoolExecutor(max_workers=20) as executor:

            for password in range(100,200):

os.walk遍历目录下所有文件路径

import os

root_path = 'C:\\Users\\PASSWORD\\Desktop\\temp'

def get_path(root_path):
    file_name = []
    for filepath,dirnames,filenames in os.walk(root_path):
        for filename in filenames:
            file_name.append(os.path.join(filepath,filename))
    return file_name

path = get_path(root_path)
for i in path:print(i)
    
# filepath表示当前正在访问的文件夹路径
# dirnames表示该文件夹下的所有子目录名
# filenames表示该文件夹下的所有文件

获取命令执行结果

import subprocess

def subprocess_popen(statement):
    p = subprocess.Popen(statement, shell=True, stdout=subprocess.PIPE)  # 执行shell语句并定义输出格式
    while p.poll() is None:  # 判断进程是否结束（Popen.poll()用于检查子进程（命令）是否已经执行结束，没结束返回None，结束后返回状态码）
        if p.wait() != 0:  # 判断是否执行成功（Popen.wait()等待子进程结束，并返回状态码；如果设置并且在timeout指定的秒数之后进程还没有结束，将会抛出一个TimeoutExpired异常。）
            print("命令执行失败")
            return False
        else:
            re = p.stdout.readlines()  # 获取原始执行结果
            result = []
            for i in range(len(re)):  # 由于原始结果需要转换编码，所以循环转为utf8编码并且去除\n换行
                res = re[i].decode('utf-8').strip('\r\n')
                result.append(res)
            return result

print(subprocess_popen('whoami'))

通过telnet扫描并匹配服务（linux下）

import warnings
warnings.filterwarnings("ignore",category=DeprecationWarning)

import telnetlib
import socket
import argparse

def check_port_with_telnet(host, port, timeout=5):
    try:
        # 创建一个 Telnet 对象
        tn = telnetlib.Telnet()
        # 连接到指定的主机和端口
        tn.open(host, port, timeout)
        tn.close()
        return True
    except (socket.timeout, ConnectionRefusedError):
        return False
    except Exception as e:
        print(f"发生错误: {e}")
        return False

def get_service_name(port):
    service_name = None
    try:
        with open('/etc/services', 'r') as f:
            for line in f:
                if line and not line.startswith('#'):
                    parts = line.split()
                    if len(parts) >= 2:
                        port_info = parts[1].split('/')[0]
                        if port_info.isdigit() and int(port_info) == port:
                            service_name = parts[0]
                            break
    except FileNotFoundError:
        print("/etc/services 文件未找到。")
    except Exception as e:
        print(f"读取 /etc/services 时发生错误: {e}")
    
    return service_name

def main():
    # 创建命令行参数解析器
    parser = argparse.ArgumentParser(description='检查指定主机和端口是否开放，并匹配服务名称。')
    parser.add_argument('host', type=str, help='主机名或IP地址')
    parser.add_argument('port', type=int, help='需要检查的端口号')

    # 解析命令行参数
    args = parser.parse_args()

    host = args.host
    port = args.port

    if check_port_with_telnet(host, port):
        print(f"端口 {port} 在 {host} 上开放。")
        service_name = get_service_name(port)
        if service_name:
            print(f"对应的服务: {service_name}")
        else:
            print("没有找到对应的服务。")
    else:
        print(f"端口 {port} 在 {host} 上关闭。")

if __name__ == "__main__":
    main()
